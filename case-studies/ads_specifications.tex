\startsubsection[title={Goal specifications}]

  This section provides the complete specifications for the goals of the
  Brussels Ambulance Despatch System, presented in \in{Section}[sec:bads].

  \subsubsubject{Achieve [Incident Resolved]}

  \placefigure[]
 	  [fig:ads_spec_1]
 	  {Refinements for \goal{Achieve [Incident Resolved]}.}
    {\externalfigure[../images/app-ads/ads_spec_1.pdf]}

  \startkaosspec
  	\GoalName {Achieve [Incident Resolved]}
  	\KaosAttribute {Def} {All incidents in the real-world shall be resolved. An incident is resolved if the victim is treated on the incident scene or transported to the hospital.}
  	\KaosAttribute {RSR} {0.95}
  	\KaosAttribute {RefinedBy} {Achieve [Incident Reported], Achieve [Ambulance On Scene When Incident Reported], Achieve [Incident Resolved When Ambulance On Scene]}
  \stopkaosspec
  
  \startkaosspec
  	\GoalName {Achieve [Incident Reported]}
  	\KaosAttribute {Def} {All incidents in the real-world shall be reported to the ambulance dispatching.}
  	\KaosAttribute {AssignedTo} {Public}
  \stopkaosspec

  \startkaosspec
  	\GoalName {Achieve [Ambulance On Scene When Incident Reported]}
  	\KaosAttribute {Def} {An ambulance shall be on scene within 12 minutes for every reported incident.}
  	\KaosAttribute {RefinedBy} {Achieve [Ambulance Mobilized When Incident Reported], Achieve [Ambulance Left Station When Mobilized], Achieve [Mobilized Ambulance On Scene]}
  \stopkaosspec

  \startkaosspec
  	\GoalName {Achieve [Incident Resolved When Ambulance On Scene]}
  	\KaosAttribute {Def} {The incident shall be resolved by the ambulance staff when the ambulance in on the incident scene.}
  	\KaosAttribute {RefinedBy} {Achieve [Incident Resolved On Scene When No Transport Required], Achieve [Patient Transported At Hospital When Required]}
  \stopkaosspec
  
  \subsubsubject{Achieve [Incident Resolved When Ambulance On Scene]}
  
  \placefigure[]
 	  [fig:ads_spec_2]
 	  {Refinements for \goal{Achieve [Incident Resolved When Ambulance On Scene]}.}
    {\externalfigure[../images/app-ads/ads_spec_2.pdf]}

  \startkaosspec
  	\GoalName {Achieve [Patient Transported At Hospital When Required]}
  	\KaosAttribute {Def} {The patient shall be transported to the nearest and most appropriate hospital when required by the patient condition. The incident is resolved when the patient is at hospital.}
  	\KaosAttribute {RefinedBy} {Maintain [Appropriate Hospital Known], Achieve [Patient Transported At Hospital When Required]}
  \stopkaosspec

  \startkaosspec
  	\GoalName {Achieve [Incident Resolved On Scene When No Transport Required]}
  	\KaosAttribute {Def} {The incident shall be resolved on the incident scene if no transport of the victim towards an hospital is required.}
  	\KaosAttribute {ObstructedBy} {Patient Not Treated At Location}
  	\KaosAttribute {AssignedTo} {Ambulance Staff}
  \stopkaosspec

  \startkaosspec
  	\GoalName {Achieve [Patient Transported At Hospital When Required]}
  	\KaosAttribute {Def} {The patient shall be transported to the indicated hospital when required by the patient condition.}
  	\KaosAttribute {ObstructedBy} {Patient Not Admitted At Or Transported To Hospital}
  	\KaosAttribute {AssignedTo} {Ambulance Staff}
  \stopkaosspec

  \startkaosspec
  	\GoalName {Maintain [Appropriate Hospital Known]}
  	\KaosAttribute {Def} {The nearest and most appropriate hospital is known by the ambulance staff. Such hospital is indicated on the mobilization order.}
  	\KaosAttribute {AssignedTo} {Ambulance Staff}
  \stopkaosspec
